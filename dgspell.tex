% Macros to define spelling-portability macros.
% Query whether a macro is already defined.
\def\SPnamexdef#1{\expandafter\xdef\csname #1\endcsname/}
\def\SPnameuse#1{\csname #1\endcsname/} % Not used.
\def\IFundefSP#1{%
 \expandafter\ifx\csname#1\endcsname\relax}

\def\defSP[#1=#2]{%
 \IFundefSP{#1}\else\message{REDEFINED SPELLING RULE: #1.}\fi%
 \SPnamexdef{#1}{#2}% Define it globally.
 }

% 2013-4-21. It took about 50 attempts to get the test in \SP to work.
% See the TeXbook page 210 for the incomprehensible explanation of this.
% I still don't have a clue why it works, or why other attempts did not work.
% TeX is still the most painful programming language in the known universe!!!
\def\SPlocaleUK{\SPlocUK}
\def\SPlocaleUS{\SPlocUS}
% Set this macro to determine the locale.
\let\SPlocale\SPlocaleUK
\def\SP#1#2#3{%
 \ifx\SPlocale\SPlocaleUS%
  \defSP[#1=#3]%        % US spelling.
 \else%
  \defSP[#1=#2]%        % UK spelling (default).
 \fi%
 }
